\subsection{黑体辐射}
使用经典理论给出公式的Wein、Rayleigh和Jeans都失败了.
Rayleigh-Jeans公式以能量按自由度均分定理为基础：
\begin{equation}
    I(\omega) = \dfrac{\omega^2}{\pi^2 c^2}kT
\end{equation}
在$\omega$较大时快速发散，即\textcolor{PurpleDeep}{\textbf{紫外灾难}}.

Planck以插值法通过Wien公式和Rayleigh-Jeans公式拟合得到了Plunck公式.
\begin{derivation}
    \[
        I(w) = \dfrac{1}{\textcolor{PurpleDeep}{e^{a\omega/kT}}-1} \cdot \dfrac{b\omega^3}{\pi^2 c^2}\quad a,b\text{为常数}
    \]
    Planck发现$\textcolor{PurpleDeep}{e^{a\omega/kT}}$形似Bolztmann分布$N_n = N_1 e^{-(E_n-E_1)/kT}$.
    
    因此，他假设\textbf{辐射场中偶极振子的能量是一份一份的，每份能量依赖于角频率}.
    \begin{equation*}
        E = a\omega \qquad N_n = N_0 e^{-E_n/kT} = N_0 e^{-na\omega/kT}
    \end{equation*}
    平均能量：
    \begin{align*}
        \langle E \rangle &= \frac{\sum_{n} N_n \cdot E_n}{\sum_{n} N_n} \\[1ex]
        &= \frac{\sum_{n} N_0 e^{-na\omega/kT} \cdot na\omega}{\sum_{n} N_0 e^{-na\omega/kT}} \\[1ex]
        &= \frac{a\omega (0 \cdot e^{-0/kT} + 1 \cdot e^{-1a\omega/kT} + 2 \cdot e^{-2a\omega/kT} + \cdots)}{e^{-0/kT} + e^{-1a\omega/kT} + e^{-2a\omega/kT} + \cdots} \\[1ex]
        &\xlongequal{\varepsilon = e^{-a\omega/kT}} \frac{a\omega (\varepsilon + 2\varepsilon^2 + 3\varepsilon^3 + \cdots)}{1 + \varepsilon + \varepsilon^2 + \varepsilon^3 + \cdots} \\[1ex]
        &= \frac{a\omega \varepsilon \frac{d}{d\varepsilon} (1 + \varepsilon + \varepsilon^2 + \varepsilon^3 + \cdots)}{1 + \varepsilon + \varepsilon^2 + \varepsilon^3 + \cdots} \\[1ex]
        &\xlongequal{\text{Taylor 展开}} \frac{a\omega \varepsilon \frac{d}{d\varepsilon} (\frac{1}{1-\varepsilon})}{\frac{1}{1-\varepsilon}} \\[1ex]
        &= a\omega \varepsilon \frac{1}{1-\varepsilon} = a\omega \frac{e^{-a\omega/kT}}{1 - e^{-a\omega/kT}} \\[1ex]
        &= a\omega \frac{1}{e^{a\omega/kT} - 1}
    \end{align*}
    Rayleigh-Jeans 公式中的能量项为$kT$，用$\langle E \rangle$取代之：
    \[
    I(\omega) = \frac{\omega^2}{\pi^2 c^2} \textcolor{PurpleDeep}{kT} = \frac{\omega^2}{\pi^2 c^2} \cdot \textcolor{PurpleDeep}{a\omega \frac{1}{e^{a\omega/kT} - 1}} = \frac{1}{e^{a\omega/kT} - 1} \frac{a\omega^3}{\pi^2 c^2}
    \]
    这就是Plunck公式，式中$a$即为如今的$\hbar $.
\end{derivation}

\section{Schrödinger方程}
\subsection{Schrödinger方程的推导}
\begin{derivation}
    Schrödinger根据光的波动性和de Brogile的波粒二象性猜测出了实物粒子的波动方程：
    \[\varphi(x,t) = A\cos(kx-\omega t)\]
    根据Euler公式$e^{i\theta} = \cos\theta + i\sin\theta$有：
    \[\varphi(x,t) = A\text{Re}\{e^{i(kx-\omega t)}\}\]
    为了便于数学处理，用整个复数表示波：
    \[\varphi(x,t) = Ae^{i(kx-\omega t)}\]
    实物粒子$E = \hbar\omega \Rightarrow k = \frac{2\pi}{\lambda} = \frac{p}{\hbar}$代入上式中，于是有自由实物粒子的波动方程：
    \[\psi(x,t) = Ae^{\frac{i}{\hbar}(px-Et)}\]
    考察波函数关于t的变化：
    \[\frac{\partial}{\partial t}\psi = -\frac{i}{\hbar}EAe^{\frac{i}{\hbar}(px-Et)} \Rightarrow \textcolor{PurpleDeep}{i\hbar\frac{\partial}{\partial t}\psi=E\psi}\]
    考察波函数关于x的变化，为了获取能量项，取二阶导：
    \[\frac{\partial^2}{\partial x^2}\psi = \left(-\frac{p}{i\hbar}\right)^2 Ae^{\frac{i}{\hbar}(px-Et)} = \left(-\frac{1}{i\hbar}\right)^2 2\mu E\psi \Rightarrow \textcolor{PurpleDeep}{\frac{1}{2\mu}\left(-i\hbar\frac{\partial}{\partial x}\right)^2 \psi = E\psi}\]
    联立上述二式：
    \[i\hbar\frac{\partial}{\partial t}\psi = \frac{1}{2\mu}\left(-i\hbar\frac{\partial}{\partial x}\right)^2 \psi = E\psi\]
    这就是Schrödinger方程
\end{derivation}
事实上，Schrödinger作上述推导\textbf{纯粹基于猜想}，没有严密的数学推导，它在量子力学中是作为公理存在的.

\subsection{Born统计诠释}
\begin{center}
    \textit{波函数$\psi$的意义是什么？}
    
    \textit{到底是什么在波动？}
\end{center}

Born类比光波的模平方等于光强，即光子在该处的密度大小，故而对实物粒子，\textbf{波函数的模平方表示实物粒子出现在该处的概率密度大小}

因此有\textbf{归一化条件}：
\[\int_{-\infty }^{+\infty}|\psi|^2 \dd{\tau } = 1\]

而波函数本身是概率振幅，是概率的平方根，没有实际意义.

\section{电子双缝干涉实验}
\subsection{电子是如何穿过双缝的？}
电子以概率波的形式同时穿过两缝并产生干涉，落到光屏上又坍缩为粒子.
\begin{derivation}
    设电子通过缝$1$的状态为$\psi_1$，通过缝$2$的状态为$\psi_2$，这种某个量确定的态称为\textbf{本征态}.

    \noindent \textbf{若不施以观测：}

    通过双缝后的粒子处于叠加态:
    \[\psi = c_1\psi_1 + c_2\psi_2 \quad \text{(概率振幅叠加)} \qquad \text{归一化条件：}|c_1|^2 + |c_2|^2 = 1\]
    概率密度：
    \[\begin{aligned}
        \rho = |\psi|^2 = \psi^* \cdot \psi &= (C_1^* \psi_1^* + C_2^* \psi_2^*)(C_1 \psi_1 + C_2 \psi_2) \\ 
        &= |C_1|^2 |\psi_1|^2 + |C_2|^2 |\psi_2|^2 + \textcolor{PurpleDeep}{C_1^* C_2 \psi_1^* \psi_2 + C_1 C_2^* \psi_1 \psi_2^*} \\
        &= |C_1|^2 |\psi_1|^2 + |C_2|^2 |\psi_2|^2 + \textcolor{PurpleDeep}{2 \text{Re} \{ C_1^* C_2 \psi_1^* \psi_2 \}} \\
        &= |C_1|^2 |\psi_1|^2 + |C_2|^2 |\psi_2|^2 + \textcolor{PurpleDeep}{2 \text{Re} \{ |C_1| |\psi_1| e^{-i\theta_1} \cdot |C_2| |\psi_2| e^{i\theta_2} \}} \\
        &= |C_1|^2 |\psi_1|^2 + |C_2|^2 |\psi_2|^2 + \textcolor{PurpleDeep}{2 |C_1| |\psi_1| |C_2| |\psi_2| \cos(\theta_2 - \theta_1)} \\
        &= |C_1|^2 |\psi_1|^2 + |C_2|^2 |\psi_2|^2 + \textcolor{PurpleDeep}{\underbrace{2 |C_1| |\psi_1| |C_2| |\psi_2| \cos \theta}_{\text{干涉项}}}
    \end{aligned}\]

    \noindent \textbf{若施以观测：}

    粒子在经过探测器后发生坍缩，落入某个本征态，干涉花样消失.
    \begin{align*}
    \psi = C_1\psi_1 + C_2\psi_2
    \begin{cases} 
        \text{以 } |C_1|^2 \text{ 的概率坍缩到 } \psi_1 \\
        \text{以 } |C_2|^2 \text{ 的概率坍缩到 } \psi_2
    \end{cases} \\
    \text{总强度}\quad I = I_1 + I_2 = \underbrace{|C_1|^2 |\psi_1|^2 + |C_2|^2 |\psi_2|^2}_{\text{无干涉项}}
    \end{align*}
\end{derivation}
以上内容被称为\textbf{态叠加原理/测量假设}.

实验现象表明：

\begin{itemize}
\item \textbf{一旦确定了电子从哪个缝通过，就会破坏掉干涉花样.}
    \begin{mistake}
        注意，“观测”不是一个唯心的过程，而是粒子的位置信息与外部环境发生纠缠的过程，不以人的意志为转移.
    \end{mistake}
\item \textbf{电子路径的精确度越大，干涉花样的精确度越小.} \\
这是一个粗略版的\overtext{不确定性原理}{uncertainty principle}.
\end{itemize}

\subsection{延迟选择实验}
\overtext{延迟选择}{delay-selection}实验表明，在电子通过狭缝后再加上探测器仍会使干涉花样消失.

\section{Bohr原子模型}
Bohr之前，有J.J.Thompson的\textbf{枣糕模型}和Rutherford的\textbf{行星模型}.
早在1915年，Schrödinger方程提出之前(1925)，Bohr创新性地基于Rutherford行星模型、Planck能量分立假说、Einstein光子说和氢光谱经验公式$\tilde{v}(m,n) = R_{\mathrm H}\left(\frac{1}{m^2}-\frac{1}{n^2}\right)$提出了\textbf{Bohr原子模型}

\subsection{Bohr原子模型的三大假设}
\begin{itemize}
    \item \textbf{定态假设}\\
    电子绕核作圆周运动时只能在一些分立的轨道上，并具有稳定的能量，即不向外辐射电磁波.
    \item \textbf{辐射条件}\\
    当原子中的电子从一个定态n跃迁至定态m时会以电磁波的形式放出/吸收能量.
    \item \textbf{角动量量子化}
    \begin{gather*}
        L = m_e vr = n\hbar \Rightarrow E_n = -\frac{hcR_{\mathrm{H}}}{n^2} \\
        h\nu_{m \rightarrow n} = E_n - E_m \Rightarrow \nu_{m \rightarrow n} = cR_{\mathrm H}\left(\frac{1}{m^2}-\frac{1}{n^2}\right)
    \end{gather*}
\end{itemize}

\subsection{Bohr原子模型的缺陷}
通过强行假定克服能量不守恒的问题，且只适用于氢原子与类氢原子，对多电子体系无效.

\section{矩阵力学}
矩阵力学是由Heisenberg独立地提出的量子力学等价形式.

\subsection{实证主义}
Heisenberg主张整个量子力学应以\overtext{可观测量/力学量}{observable}为基础.

Bohr理论的“轨道”不具备物理意义，而能级差是可观测的，有物理意义.

描述能级差需基于初、末两个状态共同决定，描述一个状态需要一系列数，因此可观测量应当是一个二维方阵，表达了状态的变化.

\subsection{矩阵与算符}
\begin{itemize}
    \item \textbf{从具象视角来看：} \\
    Heisenberg利用\overtext{矩阵}{matrix}表达状态的改变方式，用一维列矩阵表示一个态.
    \item \textbf{从抽象视角来看：} \\
    矩阵作为\overtext{算符}{operator}，对应了一种观测(操作)，作用是使一个态转换成另一个态.
    \begin{gather*}
        \begin{bmatrix} \hat{O} \end{bmatrix} 
        \begin{bmatrix} \psi \end{bmatrix} = 
        \begin{bmatrix} \psi' \end{bmatrix} \qquad
        \text{可记作 } \hat{O} |\psi\rangle = |\psi'\rangle
    \end{gather*}
    这是一种线性代数的语言.
\end{itemize}

\section{线性代数基础}
\subsection{线性矢量空间(LVS)}
\begin{definition}
    \overtext{线性矢量空间}{linear vector space} $\mathbb{V}$ 是由众多矢量$\ket{v_i}$构成的集合.
\end{definition}
LVS满足如下法则：
\begin{enumerate}
    \item \textbf{封闭性} \\
    矢量空间中任意矢量的线性组合仍在该矢量空间中.
    \item \textbf{数乘的分配律}
    \begin{gather*}
        a ( |v_1\rangle + |v_2\rangle ) = a |v_1\rangle + a |v_2\rangle \\
        (a + b) |v\rangle = a |v\rangle + b |v\rangle
    \end{gather*}
    \item \textbf{矢量和交换律}
    \[|v_1\rangle + |v_2\rangle = |v_2\rangle + |v_1\rangle\]
    \item \textbf{矢量和结合律}
    \[|v_1\rangle + ( |v_2\rangle + |v_3\rangle ) = ( |v_1\rangle + |v_2\rangle ) + |v_3\rangle\]
    \item \textbf{零矢量存在性} \\
    存在一个零矢量对任意矢量都满足$|0\rangle + |v\rangle = |v\rangle$，即$|0\rangle = 0 |v\rangle$.
    \item \textbf{逆矢量存在性} \\
    对任意矢量$\ket{v}$都存在$\ket{-v}$使$|v\rangle + |-v\rangle = |0\rangle$，即$|-v\rangle = - |v\rangle$.
\end{enumerate}

反直觉的是，如所有的$2\times 2$矩阵、定义在$(a,b)$上的所有函数$\{f\}$等“不那么矢量”的东西也满足LVS的定义.

也就是说：

\textbf{LVS中的元素不需要大小和方向.}

\begin{definition}
    构造矢量空间$\mathbb{V}$时所用到的所有可能的组合系数$a$所在的空间称为$\mathbb{V}$的\textbf{基域}.
    \[
    \mathbb{V}\text{的基域}
    \begin{cases}
        \text{由所有实数组成} \rightarrow \mathbb{V}\text{为实矢量空间} \\
        \text{由所有复数组成} \rightarrow \mathbb{V}\text{为复矢量空间}
    \end{cases}
    \]
\end{definition}

\begin{definition}
    线性相关性
\end{definition}

\begin{definition}
    若一个LVS最多容纳$n$个线性无关的矢量，则称该空间为$n$维的.
\end{definition}

\begin{definition}
    将$n$维空间中的$n$个线性无关的矢量称作该矢量空间的一组\overtext{基}{base}.
\end{definition}
一般来说，基不是唯一的.

\begin{definition}
    矢量在基下的展开系数称为其在这组基下的\overtext{分量}{component}.
\end{definition}
对于给定的基，同一矢量的分量是唯一的.

\subsection{内积空间(IPS)}
传统的矢量内积$\vec{A} \cdot \vec{B} = |\vec{A}| |\vec{B}| \cos\theta = \sum_{i}^n a_i b_i$具有如下性质：
\begin{itemize}
    \item $\vec{A} \cdot \vec{B} = \vec{B} \cdot \vec{A}$
    \item $\vec{A} \cdot \vec{A} \ge 0$
    \item $\vec{A} \cdot (b\vec{B} + c\vec{C}) = b\vec{A} \cdot \vec{B} + c\vec{A} \cdot \vec{C}$
\end{itemize}

\begin{extension}{Dirac符号}
    记$|v\rangle = \begin{bmatrix} v_1 \\ v_2 \\ v_3 \\ \vdots \\ v_n \end{bmatrix} = \sum_{i}^n v_i |v_i\rangle$,$n$为LVS的维度.\\[1em]
    记矢量$\ket{A}$和矢量$\ket{B}$的内积$\langle A|B \rangle$.\\
    内积应当是个数，因此$\bra{A}$应为一个行矩阵.\\[1em]
    $
    \text{称}
    \begin{cases}
        \ket{~} \text{为\overtext{右矢}{ket}，其在一组基下的具象是一个列矩阵.}\\
        \bra{~} \text{为\overtext{左矢}{bra}，其在一组基下的具象是一个行矩阵.}
    \end{cases}
    $
\end{extension}

态矢的内积应满足如下性质：
\begin{itemize}
    \item $\langle v|v\rangle \ge 0$矢量的模长是个非负实数
    \item $\langle A|B\rangle = \langle B|A\rangle^*$在实矢量空间中退化为$\langle A|B\rangle = \langle B|A\rangle$
    \item $\langle A|B\rangle = \sum_{i}^n a_i^* b_i = [a_1^* \ a_2^* \ a_3^* \ \dots \ a_n^*] \begin{bmatrix} b_1 \\ b_2 \\ b_3 \\ \vdots \\ b_n \end{bmatrix}$
\end{itemize}

综上，$\bra{v}$在形式上是$\ket{v}$的转置复共轭，称为\overtext{厄米共轭}{Hermitian conjugate}，用“\overtext{$\dagger$}{dagger}”表示，记作:
\[(|v\rangle)^\dagger = \langle v|\]

\begin{extension}{态矢的线性性}
    有：
    \[\langle v_1 | (a |v_2 \rangle + b |v_3 \rangle) = \langle v_1 | a v_2 + b v_3 \rangle = a \langle v_1 | v_2 \rangle + b \langle v_1 | v_3 \rangle\]
    则：
    \[
    \begin{aligned} \langle a v_2 + b v_3 | v_1 \rangle &= [\langle v_1 | (a |v_2 \rangle + b |v_3 \rangle)]^* \\ &= a^* \langle v_1 | v_2 \rangle^* + b^* \langle v_1 | v_3 \rangle^* \\ &= a^* \langle v_2 | v_1 \rangle + b^* \langle v_3 | v_1 \rangle \\ &= (a^* \langle v_2 | + b^* \langle v_3 |) | v_1 \rangle \end{aligned}
    \quad \begin{aligned}
        & \langle a v_2 + b v_3 | = a^* \langle v_2 | + b^* \langle v_3 | \\
        \Rightarrow \quad &\text{即} \langle av | = a^* \langle v |\\
        &\text{这保证了} \langle cA | cA \rangle \text{为实数}
    \end{aligned}
    \]
\end{extension}

\begin{definition}
    满足所有内积性质的LVS称为\overtext{内积空间}{inner product space}.
\end{definition}
\begin{definition}
    若$\langle v_1|v_2 \rangle = 0$，称$\ket{v_1}$与$\ket{v_2}$是正交的.
\end{definition}
\begin{definition}
    将$\sqrt{\langle v|v \rangle}$称作$\ket{v}$的范数.若$\ket{v}$的范数为$1$，则称$\ket{v}$为归一化的矢量.
\end{definition}
\begin{definition}
    一组范数为$1$且两两正交的基称为\textbf{正交归一基}.

\end{definition}
对一组线性无关的基，总可以通过线性组合构造出一组正交归一基.

对正交归一基，有：
\[\langle v_i | v_j \rangle = \delta_{ij} = \begin{cases}
    0,\quad i \neq j \\
    1,\quad i = j
\end{cases}\]
其中$\delta_{ij}$称为\textbf{Kronecker delta符号}.

\subsection{正交归一基}
\begin{question}
    已知一给定的矢量和一组正交归一基，如何确定其展开系数？
\end{question}
\begin{solution}
    即$|v\rangle = \sum_{i}^n v_i |v_i\rangle \quad \text{求} v_i$. \\
    两边同时左乘$\ket{v_j}$：
    \[\langle v_j | v \rangle = \sum_{i}^n v_i \langle v_j | v_i \rangle = \sum_{i}^n v_i \delta_{ij} = v_j\]
    因此$v_i = \langle v_i | v \rangle$，进而有：
    \begin{align*}
        |v\rangle &= \sum_{i}^n \textcolor{GreenDeep}{v_i} |v_i\rangle \\
        &= \sum_{i}^n \textcolor{GreenDeep}{\langle v_i | v \rangle} |v_i\rangle \\
        &= \textcolor{GreenDeep}{\sum_{i}^n |v_i\rangle \langle v_i |} v \rangle
    \end{align*}
    综上，称$\sum_{i}^n |v_i\rangle \langle v_i | \equiv \hat{I}$为\overtext{恒等算符}{indentity operator}：它把一个矢量拆成基的分量，再加回来仍是原矢量.

    称$\hat{P} \equiv |v_i\rangle \langle v_i |$为$\ket{v_i}$的\overtext{投影算符}{projection operator}：
    \[\hat{P} |v\rangle = |v_i\rangle \langle v_i | v \rangle = v_i |v_i\rangle\]
    $\hat{P}$挑选出了$\ket{v}$中沿着$\ket{v_i}$的部分.
\end{solution}
\begin{question}
    如何用$n$个线性无关的矢量构造一组正交归一基？
\end{question}
\begin{extension}{Schmidt正交化}
    三个线性无关矢量$\ket{1},\ket{2},\ket{3}$.
    \begin{enumerate}
        \item 将$\ket{1}$放缩，使之归一化为$\ket{\text{I}}$，有$\braket{\text{I}}{\text{I}}=1$\\
        若$\ket{2}$与$\ket{1}$不正交，则$\braket{1}{2}\neq 0$，$\ket{2}$中仍含有与$\ket{1}$平行的分量，大小为$\braket{1}{2}$.
        \item 减去$\ket{2}$中与$\ket{1}$平行的分量
        $$|2'\rangle = |2\rangle - \hat{P}_{\text{I}} |2\rangle = |2\rangle - |\text{I}\rangle \langle \text{I} | 2 \rangle$$
        再放缩成$\ket{\text{II}}$使之归一化.
        \item 将$\ket{3}$中与$\ket{\text{I}},\ket{\text{II}}$平行的分量减去
        $$|3'\rangle = |3\rangle - \hat{P}_{\text{I}} |3\rangle - \hat{P}_{\text{II}} |3\rangle$$
        再放缩成$\ket{\text{III}}$使之归一化.
    \end{enumerate}
    则$\ket{\text{I}},\ket{\text{II}},\ket{\text{III}}$为一组正交归一基.
\end{extension}

\subsection{线性算符}
算符的作用是使一个矢量变为另一个矢量.

\[
\hat{O}\ket{v} = \ket{v'} \qquad
\begin{aligned}
    &\text{由于$\ket{v}$和$\ket{v'}$都是$n$维的列矩阵}\\
    &\text{故而$\hat{O}$的具象应当是一个$n\times n$方阵}
\end{aligned}
\]
\begin{definition}
    线性算符即满足$\begin{cases}
        \hat{O} \alpha |v\rangle = \alpha \hat{O} |v\rangle \\
        \hat{O}(\alpha |v_1\rangle + \beta |v_2\rangle) = \alpha \hat{O} |v_1\rangle + \beta \hat{O} |v_2\rangle
    \end{cases}$的算符.
\end{definition}
\begin{definition}
    矩阵一般是不对易的，即$\hat{O}\hat{\Omega}\neq\hat{\Omega}\hat{O}$.称$[\hat{O},\hat{\Omega}]=\hat{O}\hat{\Omega}-\hat{\Omega}\hat{O}$为$\hat{O}$与$\hat{\Omega}$的\overtext{对易子}{commutator}.
\end{definition}

算符的具体作用方式可以从两个方面来理解：
\[
\hat{O} |v\rangle =
\begin{cases}
    |v'\rangle = \sum_{i} v'_i |v_i\rangle & \text{分量变，基不变} \\
    \hat{O} \sum_{i} v_i |v_i\rangle = \sum_{i} v_i \hat{O} |v_i\rangle = \sum_{i} v_i |v'_i\rangle & \text{基变，分量不变}
\end{cases}
\]

对于矩阵形式：
\[
\begin{aligned}
    v'_i &= \langle v_i | v' \rangle = \langle v_i | \hat{O} | v \rangle\\
    &= \langle v_i | \hat{O} \sum_{j} v_j | v_j \rangle \\
    &= \sum_{j} \langle v_i | \hat{O} | v_j \rangle v_j \qquad \langle v_i | \hat{O} | v_j \rangle \text{可以挑选出矩阵$\hat{O}$中第$i$行$j$列的元素} \\
    &= \sum_{j} O_{ij} v_j \qquad \text{这就是矩阵乘法的操作方法}
\end{aligned}
\]

考察$\ket{\alpha}\bra{\beta}$的含义：
\[
|\alpha\rangle \langle\beta| = \begin{bmatrix} \alpha_1 \\ \alpha_2 \\ \vdots \\ \alpha_n \end{bmatrix} \begin{bmatrix} \beta_1^* & \beta_2^* & \dots & \beta_n^* \end{bmatrix} = \begin{bmatrix} \alpha_1 \beta_1^* & \alpha_1 \beta_2^* & \dots & \alpha_1 \beta_n^* \\ \alpha_2 \beta_1^* & \alpha_2 \beta_2^* & \dots & \alpha_2 \beta_n^* \\ \vdots & \vdots & \ddots & \vdots \\ \alpha_n \beta_1^* & \alpha_n \beta_2^* & \dots & \alpha_n \beta_n^* \end{bmatrix}
\]
显然，$\ket{~}\bra{~}$构造了一个算符,将这种运算称为矢量的\textbf{外积}.
\begin{extension}{算符的性质}
    $(\hat{A} \hat{B})^T = \hat{B}^T \hat{A}^T \qquad (\hat{A} \hat{B})^\dagger = \hat{B}^\dagger \hat{A}^\dagger \qquad (\hat{O} |\alpha\rangle)^\dagger = \langle\alpha| \hat{O}^\dagger \text{即} |\hat{O} \alpha\rangle^\dagger = \hat{O}^\dagger \langle\alpha|$ \\
    若存在$\hat{O} |v\rangle = |v'\rangle,\hat{O}^{-1} |v'\rangle = |v\rangle$，则称$\hat{O}^{-1}$为$\hat{O}$的逆，满足$\hat{O}^{-1} \hat{O} |v\rangle = \hat{I} |v\rangle$.
\end{extension}

\subsection{厄米算符与反厄米算符}
\begin{definition}
    \begin{itemize}
        \item \overtext{厄米算符}{Hermite operator}：满足$\hat{O}^{\dagger} = \hat{O},O_{ij} = O^*_{ji}$的算符.
        \item \overtext{反厄米算符}{Anti-Hermite operator}：满足$\hat{O}^{\dagger} = -\hat{O},O_{ij} = -O^*_{ji}$的算符.
    \end{itemize}
\end{definition}

\begin{theorem}
    任意算符都可以写成厄米算符和反厄米算符的线性组合.
    \[\hat{\Omega} = \underbrace{\frac{\hat{\Omega}+\hat{\Omega}^{\dagger}}{2}}_{\text{厄米算符}} + \underbrace{\frac{\hat{\Omega}-\hat{\Omega}^{\dagger}}{2}}_{\text{反厄米算符}} \]
\end{theorem}

两个厄米算符的和一定是厄米算符.

两个厄米算符的积不一定是厄米算符，除非它们是对易的.

\subsection{幺正算符}
\begin{definition}
    \overtext{幺正算符}{unitary operator}：满足$\hat{U}^{\dagger} \hat{U} = \hat{I}$，即$\hat{U}^{\dagger} = \hat{U}^{-1}$的算符.
\end{definition}

幺正变换是一种抽象的旋转操作，它是保模长、保内积的.
$$\langle \alpha' | \beta' \rangle = \langle \hat{U} \alpha | \hat{U} \beta \rangle = \langle \alpha | \hat{U}^\dagger \hat{U} | \beta \rangle = \langle \alpha | \beta \rangle$$

两个幺正算符的积仍是幺正算符，因为两次旋转仍是旋转.

\subsection{本征值与本征矢}
\begin{definition}
    若算符$\hat{O}$作用在某一矢量上的结果等价于将该矢量放缩一个倍数，则称该矢量为$\hat{O}$的本征矢，该倍数称为$\hat{O}$的本征值.
    \[\hat{O}\ket{n}=O_n \ket{n}\text{称为$\hat{O}$的本征方程}\]
\end{definition}

\noindent\textbf{本征方程的求解：}
\[(\hat{O}-O_n \hat{I}) \ket{n} = \ket{0}\]
根据Cramer法则，其有非零解的条件为：
\[\det(\hat{O}-O_n \hat{I})=0 \quad \text{(久期方程)}\]
\noindent\textbf{本征值与本征矢具有如下性质：}
\begin{itemize}
    \item 本征矢模长的绝对大小没有意义.
    \item 对角矩阵和三角矩阵对角线上的值即为其本征值.
    \item 有时同一个本征值会对应多个本征矢，这种情况称作\overtext{简并}{degenerate}.\\[1em]
    如$\begin{bmatrix} 1 & 0 & 0 \\ 0 & 1 & 0 \\ 0 & 0 & 3 \end{bmatrix}$的本征值为$\lambda_1 = \lambda_2 = 1, \lambda_3 = 3$.

    将$1$代回本征方程：
    \[
    \begin{bmatrix} 0 & 0 & 0 \\ 0 & 0 & 0 \\ 0 & 0 & 2 \end{bmatrix} \begin{bmatrix} \alpha \\ \beta \\ \gamma \end{bmatrix} = 0 \Rightarrow \begin{matrix} \alpha, \beta \text{ 取任意值} \\ \gamma = 0 \end{matrix}
    \qquad\qquad \text{取} |1\rangle = \begin{bmatrix} 1 \\ 0 \\ 0 \end{bmatrix}, \quad |2\rangle = \begin{bmatrix} 0 \\ 1 \\ 0 \end{bmatrix}
    \]

    $\ket{1},\ket{2}$都对应本征值$1$，称为\textbf{二重简并}.

    称由所有$\begin{bmatrix} \alpha \\ \beta \\ 0 \end{bmatrix}$构成的二维LVS为$\lambda = 1$时的\textbf{本征子空间}.
\end{itemize}

\subsection{厄米算符和幺正算符的本征值问题}
\begin{theorem}
    厄米算符的本征值是实数.
\end{theorem}
\begin{proof}
    设$\hat{H}$为厄米算符：
    \[\hat{H}\ket{n}=h_n\ket{n}\]
    两边同时左乘$\bra{n}$：
    \[
    \begin{aligned}
        \langle n|\hat{H}|n\rangle &= \langle n|h_n|n\rangle \\
        (\langle n|\hat{H}|n\rangle)^\dagger &= (\langle n|h_n|n\rangle)^\dagger \\
        \langle n|\hat{H}^\dagger|n\rangle &= h_n^*\langle n|n\rangle
    \end{aligned}
    \]
    由厄米性，有$\hat{H}^\dagger = \hat{H}$：
    \[h_n\langle n|n\rangle = h_n^*\langle n|n\rangle \Rightarrow h_n = h_n^*\]
    故而$h_n$为实数.
\end{proof}
\begin{theorem}
    对应厄米算符不同本征值的本征矢相互正交.
\end{theorem}
\begin{proof}
设有$h_n,h_m$是$\hat{H}$的本征值：
    \begin{align*}
        \hat{H}|n\rangle &= h_n|n\rangle \\
        \langle m|\hat{H}|n\rangle &= \langle m|h_n|n\rangle \\
        \intertext{由$\langle m|\hat{H} = (\hat{H}|m\rangle)^\dagger = (h_m|m\rangle)^\dagger = h_m\langle m|$：}
        h_m\langle m|n\rangle &= h_n\langle m|n\rangle\\
        (h_m - h_n)\langle m|n\rangle &= 0
    \end{align*}
    当$h_m \neq h_n$时，$\braket{m}{n}=0$.
\end{proof}

\begin{theorem}
    幺正算符的本征值是模长为1的复数.
\end{theorem}
\begin{proof}
    \[
    \begin{cases}
        \hat{U}\ket{n} = U_n \ket{n} \\
        \bra{n}\hat{U}^{\dagger} = \hat{U}^*_n \bra{n}
    \end{cases}
    \Rightarrow
    \begin{aligned}
        &\bra{n}\hat{U}^{\dagger}\hat{U}\ket{n} = U^*n U_n \braket{n}{n} \\
        &\braket{n}{n} = |U_n|^2 \braket{n}{n}
    \end{aligned}
    \Rightarrow |U_n|^2 = 1 \qquad U_n = e^{i\theta_n}
    \]
\end{proof}

\subsection{厄米算符的对角化}
\begin{theorem}
    算符在用其自身的一组正交归一本征矢构成的基下表示为一个对角矩阵.
\end{theorem}
\begin{proof}
    $\ket{m},\ket{n}$为$\hat{O}$的本征矢：
    \[
    \begin{aligned}
        O_{mn} &= \bra{m}\hat{O}\ket{n} = O_n \braket{m}{n} \\
        &=\begin{cases}
            O_n, &m=n \\
            0, &m \neq n
        \end{cases}
    \end{aligned}
    \]
    因此非对角线上的元素全为$0$，矩阵是对角阵.
\end{proof}

\begin{theorem}
    若$\hat{O}$在某正交归一基下是一个厄米矩阵，则总有一个幺正矩阵$\hat{U}$，使$\hat{U}^{\dagger}\hat{O}\hat{U}$在该基下是对角矩阵.
\end{theorem}
\begin{proof}
    原命题可以拆分为以下两个命题：
    \[
    \begin{cases}
        \text{\textbf{命题I：将一组正交归一基转变为另一组正交归一基的变换是幺正变换.}} \\
        \text{\textbf{命题II：厄米算符的本征矢可以构成一组正交归一基.}}
    \end{cases}
    \]

    \rule{\textwidth}{0.4pt}

    \textbf{证明命题I：}
    
    即某$n$维空间中存在两组正交归一基$\{\ket{a}\}$和$\{\ket{b}\}$.

    则存在$\hat{U}^{\dagger}\hat{U}=\hat{I}$使$\hat{U}\ket{a^{(i)}}=\ket{b^{(i)}},i=1,2,3,\cdots ,n$.

    \textbf{断言}：$\hat{U} = \sum_k |b^{(k)}\rangle \langle a^{(k)}|$

    观察$\hat{U}$作用在$\ket{a^{(i)}}$上的效果：
    \[
    \begin{aligned}
        \hat{U}|a^{(i)}\rangle &= \sum_k |b^{(k)}\rangle \langle a^{(k)}| a^{(i)}\rangle \\
        &= \sum_k |b^{(k)}\rangle \delta_{ki} \\
        &= |b^{(i)}\rangle \quad \text{发现 $\hat{U}$ 满足变换的要求}
    \end{aligned}
    \]
    接着证明该变换的幺正性：
    \[
    \begin{aligned}
        \hat{U}^\dagger \hat{U} &= \sum_k |a^{(k)}\rangle \langle b^{(k)}| \sum_{k'} |b^{(k')}\rangle \langle a^{(k')}| \\
        &= \sum_k \sum_{k'} |a^{(k)}\rangle \textcolor{GreenDeep}{\langle b^{(k)} | b^{(k')}\rangle} \langle a^{(k')}| \\
        &= \sum_k (\sum_{k'} |a^{(k)}\rangle \textcolor{GreenDeep}{\delta_{kk'}} \langle a^{(k')}|) \\
        &= \sum_k |a^{(k)}\rangle \langle a^{(k)}|\\
        &= \hat{I} \quad \text{故而 $\hat{U}$ 是幺正的}
    \end{aligned}
    \]

    \rule{\textwidth}{0.4pt}

    \textbf{证明命题II：}

    厄米算符$\hat{\Omega}$至少有一个本征值$\lambda_1$对应本征矢$\ket{\lambda_1}$.

    所有与$\ket{\lambda_1}$正交的矢量构成的子空间应为$n-1$维，记作$\mathbb{V}^{n-1}_{\perp}$.

    选取$\ket{\lambda_1}$和$\mathbb{V}^{n-1}_{\perp}$中任意$n-1$个正交归一矢量为正交归一基.

    在该基下，$\hat{\Omega}$表示为：
    \[
    \begin{bmatrix}
        \lambda_1 & 0 & 0 & \cdots & 0 \\
        0 & \Omega_{22} & \Omega_{23} & \cdots & \Omega_{2n} \\
        0 & \Omega_{32} & \Omega_{33} & \cdots & \Omega_{3n} \\
        \vdots & \vdots & \vdots & \ddots & \vdots \\
        0 & \Omega_{n2} & \Omega_{n3} & \cdots & \Omega_{nn}
    \end{bmatrix}
    \qquad
    \begin{aligned}
        &\text{由$\Omega_{ij}=\bra{i}\hat{\Omega}\ket{j}$，取$\lambda_1$和另一个基矢} \\
        &\text{有$\Omega_{1j} = \langle \lambda_1|\hat{\Omega}|j\rangle = (\hat{\Omega}|\lambda_1\rangle)^\dagger |j\rangle = \lambda_1 \langle \lambda_1|j\rangle$} \\
        &\text{因为$\ket{\lambda_1}$与其它矢量都正交} \\
        &\text{只有$j=1$时矩阵元取$\lambda_1$，其它取$0$}
    \end{aligned}
    \]

    $\hat{\Omega}$在该基下的久期方程：
    \[(\lambda_1 - \lambda_n) \cdot \det(\hat{\Omega}_{n-1} - \lambda_n \hat{I}) = 0\]
    同理$\det(\hat{\Omega}_{n-1} - \lambda_n \hat{I}) = 0$至少有一个根$\lambda_2$对应一个本征矢$\ket{\lambda_2}$且与$\ket{\lambda_1}$正交.

    重复上述推导，即可得到$n$个正交归一的本征矢.

    \rule{\textwidth}{0.4pt}

    综上，设$\hat{O}$自身的正交归一本征基为$\{\ket{b_i}\}$，在基$\{\ket{b_i}\}$上有：
    \[
    \begin{aligned}
        O_{ij} &= \langle b_i | \hat{\Omega} | b_j \rangle
        = O_j \braket{b_i}{b_j}
        = O_j \delta_{ij} \quad \text{这显然是一个对角矩阵}
    \end{aligned}
    \]
    若$\hat{O}$在正交归一基$\{\ket{a_i}\}$上是一个厄米矩阵，利用幺正变换将基$\{\ket{a_i}\}$变换到基$\{\ket{b_i}\}$上有：
    \[\langle a_i | \hat{U}^\dagger \hat{\Omega} \hat{U} | a_j \rangle = \langle b_i | \hat{\Omega} | b_j \rangle = O_j \delta_{ij}\]
    显然，在基$\{\ket{a_i}\}$上$\hat{U}^\dagger \hat{\Omega} \hat{U}$是对角阵.
\end{proof}
