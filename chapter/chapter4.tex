\chapter{一维问题}
\section{定态 Schrödinger 方程}
考虑在 $\hat{H}$不显含时的情况下求解 Schrödinger 方程：
\[i\hbar\pdv{\psi(x,t)}{t} = \hat{H}\psi(x,t) = \left[\frac{\hat{p}^2}{2\mu}+V(x)\right]\psi(x,t)\]
常采用分离变量法求解二阶齐次线性偏微分方程，令$\psi(x,t) \equiv \varphi(x) T(t)$：
\begin{align*}
    i\hbar \frac{\partial [\psi(x)T(t)]}{\partial t} &= \left[ \frac{\hat{p}^2}{2\mu} + V(x) \right] \psi(x)T(t) \\
    i\hbar \psi(x) \frac{\dd T(t)}{\dd t} &= -\frac{\hbar^2}{2\mu} \frac{\dd^2 \psi(x)}{\dd x^2} T(t) + V(x) \psi(x)T(t) \\
    \underbrace{i\hbar \frac{1}{T(t)} \frac{\dd T(t)}{\dd t}}_{\text{只与$t$相关}} &= \underbrace{-\frac{\hbar^2}{2\mu} \frac{\dd^2 \psi(x)}{\dd x^2} \frac{1}{\psi(x)} + V(x)}_{\text{只与$x$相关}}
\end{align*}
两边都会变化，而自变量却不是同一个，说明两边都只能等于一个常数.从量纲上看，这个常数显然为能量.
\[i\hbar\frac{1}{T(t)}\dv{T(t)}{t} = E = -\frac{\hbar^2}{2\mu}\dv[2]{\varphi(x)}{x}\frac{1}{\varphi(x)} + V(x)\]
\begin{align*}
    \text{左边} \qquad i\hbar\frac{1}{T(t)}\dv{T(t)}{t} &= E \\
    \frac{1}{T}\dd{T} &= \frac{E}{i\hbar}\dd{t} \\
    \ln T &= \frac{Et}{i\hbar} + C \\
    T &= A e^{-\frac{iEt}{\hbar}}
\end{align*}
\begin{align*}
    \text{右边} \qquad -\frac{\hbar^2}{2\mu}\dv[2]{\varphi(x)}{x}\frac{1}{\varphi(x)} + V(x) &= E \\
    \left[-\frac{\hbar^2}{2\mu}\dv[2]{}{x}+V(x)\right]\varphi(x) &= E\varphi(x) \\
    \hat{H}\varphi(x) &= E\varphi(x) \qquad \text{x表象下的能量本征方程}
\end{align*}
$\hat{H}\varphi(x) = E\varphi(x)$的解即为能量本征态，又称作定态，该方程也称作定态 Schrödinger 方程.
综上：\[\psi(x,t) = A\varphi(x) e^{-iEt/\hbar}\]
注意到：\[\hat{U}(t)\varphi(x) = e^{-i\hat{H}t/\hbar}\varphi(x)=e^{-iEt/\hbar}\varphi(x)\]
故而：\[\psi(x) = \psi(x,0)\]

\subsection{定态}
\begin{itemize}
    \item 定态是能量的本征态.
    \item $\varphi(x) = \psi(x,0), \quad \hat{U}(t)\varphi(x)=\psi(x,t)$.
    \item 定态的线性组合$\begin{cases}
        \text{关于$E$简并$\rightarrow$仍是定态} \\
        \text{关于$E$不简并$\rightarrow$不再是定态}
    \end{cases}$

    \begin{proof}
    \begin{itemize}
        \item \textbf{简并}：

        假设$\begin{cases}
            \hat{H}\varphi_1(x)=E_1 \varphi_1(x) \\
            \hat{H}\varphi'_1(x)=E_1 \varphi'_1(x)
        \end{cases}$

        则$\hat{H}\left[c_1\varphi_1(x) + c_2\varphi'_1(x)\right]=E_1\left[c_1\varphi_1(x) + c_2\varphi'_1(x)\right]$.

        显然$c_1\varphi_1(x) + c_2\varphi'_1(x)$也满足定态 Schrödinger 方程，于是它也是定态.

        \item \textbf{不简并}：
        
        假设$\begin{cases}
            \hat{H}\varphi_1(x)=E_1 \varphi_1(x) \\
            \hat{H}\varphi_2(x)=E_2 \varphi_2(x)
        \end{cases}\quad E_1 \neq E_2$

        则$\hat{H}\left[c_1\varphi_1(x) + c_2\varphi_2(x)\right]=E_1 c_1 \varphi_1(x) + E_2 c_2 \varphi_2(x)$.

        于是$c_1\varphi_1(x) + c_2\varphi_2(x)$不是定态.
    \end{itemize}
    \end{proof}

    \item 任意不含时力学量在定态上的几率分布和期望值不随时间变化.
    \begin{proof}
        设力学量 $\hat{\Omega}$ 的本征值为 $\omega$，对应本征态 $\ket{\omega}$.
        \begin{align*}
            \rho(\omega) &= \abs{\braket{\omega}{\psi(t)}}^2 \\
            &= \abs{\mel{\omega}{e^{-i\hat{H}t/\hbar}}{\psi(0)}}^2 \\
            &= \abs{\mel{\omega}{e^{-iEt/\hbar}}{\psi(0)}}^2 \\
            &= \abs{\braket{\omega}{\psi(0)}}^2 \underbrace{\abs{e^{-iEt/\hbar}}^2}_{\text{幺正算符的模平方为$1$}} \\
            &= \abs{\braket{\omega}{\psi(0)}}^2 \quad \text{不随时间变化}
        \end{align*}
        由此$\ev{\hat{\Omega}}=\int^{+\infty}_{\infty}\omega\rho(\omega)\dd{x}$不随时间变化.

        特别的，有$\rho(x)=\abs{\psi(x,t)}^2=\abs{\varphi(x) e^{-iEt/\hbar}}^2 = \abs{\varphi(x)}^2$不随时间变化.
    \end{proof}
\end{itemize}

\section{Virial定理和F-H定理}
\subsection{Virial定理}
在一维定态上考虑$\ev{x\cdot \hat{p}_x}$随时间的变化：
\begin{derivation}
    由Ehrenfest定理：
    \[\dv{t}\ev{x\cdot \hat{p}_x} = \frac{1}{i\hbar}\ev{\comm{x\cdot\hat{p}_x}{\hat{H}}}=0\]
    其中：
    \begin{align*}
        \comm{x\cdot\hat{p}_x}{\hat{H}} &= x\comm{\hat{p}_x}{\hat{H}}+\comm{x}{\hat{H}}\hat{p}_x \\
        &= x\comm{\hat{p}_x}{\frac{\hat{p}^2_x}{2\mu}+V(x)} + \comm{x}{\frac{\hat{p}^2_x}{2\mu}+V(x)}\hat{p}_x \\
        &= x\comm{\hat{p}_x}{V(x)} + \frac{1}{2\mu}\hat{p}_x\comm{x}{\hat{p}_x}\hat{p}_x + \frac{1}{2\mu}\comm{x}{\hat{p}_x}\hat{p}^2_x \\
        &= x(-i\hbar)\pdv{x}V(x) + i\hbar\frac{\hat{p}^2_x}{\mu} \\
        &= i\hbar\left[2\hat{T} - x\cdot\pdv{x}V(x)\right]
    \end{align*}
    故而：
    \[\ev{2\hat{T} - x\cdot\pdv{x}V(x)}=0 \Rightarrow 2\ev{\hat{T}}=\ev{x\cdot\pdv{x}V(x)}\]
    这就是\textbf{Virial定理}.推广到三维：
    \[2\ev{\hat{T}}=\ev{\vec{r}\cdot\nabla V(\vec{r})}\]
\end{derivation}

\subsection{Feynman-Hellman定理}
\begin{derivation}
    \begin{align*}
        \hat{H}\ket{\psi_n} &= E_n\ket{\psi_n} \\
        \intertext{两边对参量$\lambda$求导：}
        \pdv{\lambda}(\hat{H}\ket{\psi_n}) &= \pdv{\lambda}(E_n\ket{\psi_n}) \\
        (\pdv{\lambda}\hat{H})\ket{\psi_n} + \hat{H}\pdv{\lambda}\ket{\psi_n} &= (\pdv{\lambda}E_n)\ket{\psi_n} + E_n\pdv{\lambda}\ket{\psi_n} \\
        \ev{(\pdv{\lambda}\hat{H})}{\psi_n} + \ev{\hat{H}\pdv{\lambda}}{\psi_n} &= \ev{(\pdv{\lambda}E_n)}{\psi_n} + \ev{E_n\pdv{\lambda}}{\psi_n} \\
        \ev{(\pdv{\lambda}\hat{H})}{\psi_n} + E_n\ev{\pdv{\lambda}}{\psi_n} &= \pdv{\lambda}E_n\braket{\psi_n}{\psi_n} + E_n\ev{\pdv{\lambda}}{\psi_n} \\
        \ev{(\pdv{\lambda}\hat{H})}{\psi_n} &= \pdv{\lambda}E_n
    \end{align*}
    上式称作Feynman-Hellman定理.
\end{derivation}
